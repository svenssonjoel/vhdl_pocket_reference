\usepackage{mdframed}

% Center title vertically on page
\makeatletter
\renewcommand{\maketitle}{%
  \begin{titlepage}
    \centering
    \vspace*{\fill}
    {\LARGE\bfseries \@title \par}
    \vspace{1cm}
    {\large \@author \par}
    \vspace*{\fill}
  \end{titlepage}
}
\makeatother

% Redefine Shaded environment to add a frame
\let\oldShaded\Shaded
\let\endoldShaded\endShaded
\renewenvironment{Shaded}{%
  \begin{mdframed}[linewidth=0.5pt, innerleftmargin=5pt, innerrightmargin=5pt, innertopmargin=5pt, innerbottommargin=5pt, skipabove=8pt]%
  \begin{oldShaded}%
}{%
  \end{oldShaded}%
  \end{mdframed}%
}
